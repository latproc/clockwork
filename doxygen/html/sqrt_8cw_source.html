1 # Latproc isn't designed for numeric calculations, for one thing
2 # it has only integer numbers.

4 # here is a machine that is unstable if it's length is not
5 # close to the sqrt of its area. The machine trys to find
6 # an integer value but oscillates if the area is not a
7 # perfect square.
8 
9 SquareRoot MACHINE (area:10) {
10 
11  OPTION Area 0;
12  OPTION Length 0;
13  OPTION LSquared 0;
14  OPTION step 0;
15 
16  ENTER INIT { Area := area * 10000; step := Area / 2; }
17 
18  stable WHEN LSquared - Area < Length && LSquared - Area > -1 * Length;
19  low WHEN LSquared < Area;
20  high WHEN LSquared > Area;
21 
22  calculating DURING recalc_high { Length := Length - step; LSquared := Length * Length; step := step / 2; }
23  calculating DURING recalc_low { Length := Length + step; LSquared := Length * Length; step := step / 2; }
24 
25  ENTER low { IF (step == 0) {step := 1}; CALL recalc_low ON SELF }
26  ENTER high { IF (step == 0) {step := 1}; CALL recalc_high ON SELF }
27 }
