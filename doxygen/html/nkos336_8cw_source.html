1 # This is a stress test based on the machine
2 # discussed in New Kind of Science, Wolfram S., p336
3 #
4 # The test does not attempt to directly reproduce
5 # the original machine since it does not bother
6 # to process the board in stages.
7 
8 # In brief; a cell counts the number of black vs white
9 #cells in its neighbourhood, including itself;
10 
11 Cell MACHINE neighbours {
12 
13  white WHEN COUNT black FROM neighbours < 4
14  OR COUNT black FROM neighbours == 5;
15  black WHEN COUNT black FROM neighbours >=6
16  OR COUNT black FROM neighbours == 4;
17 
18  # commands used during initialisation
19  black DURING goBlack { }
20  white DURING goWhite { }
21 
22  ENTER INIT { CALL reset ON SELF; }
23  COMMAND reset {
24  x := RANDOM;
25  IF (RANDOM % 2 == 1) { CALL goBlack ON SELF } ELSE { CALL goWhite ON SELF }
26  }
27 }
