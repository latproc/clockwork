2 A device connector test
3 
4 This configuration provides a counter that continuously increments
5 as long as a device in "running". The "running" status is proveded
6 by device connector listening for input from a TCP connection. If
7 two seconds pass with no input, the device connector sets status
8 to "disconnected" and otherwise sets it to "running".
9 
10 To run device_connector, use:
11 
12  ./device_connector --host localhost --watch_property counter.count --client --property sampler.message --pattern '.*' --name sampler --no_timeout_disconnect
13 
14 Using this configuration, apart from the status messages on the 'sampler' machine
15 getting updated, any data sent from the monitored device is copied to the
16 'message' property in the 'sampler' machine.
17 
18 */

22  OPTION port 7711;
23  MONITORS counter;
24  PUBLISHER;

27 # This program monitors a device to detect an input.
28 # when the input comes a process is started.
29 # when the input goes off the process is stopped.
30 
31 Counter MACHINE {
32  OPTION VALUE 0;
33 
34  running WHEN SELF IS NOT idle, EXECUTE count WHEN TIMER > 1000;
35  idle INITIAL;
36 
37  counting DURING count { VALUE := VALUE + 1; }
38  COMMAND reset { VALUE := 0; SET SELF TO idle; }
39 
40  TRANSITION idle TO running ON start;
41  TRANSITION ANY TO idle ON stop;
42 }

48 Device MACHINE process {
49  OPTION status "unknown";
50  on WHEN status == "running";
51  off DEFAULT;
52  ENTER on { SEND start TO process; }
