3 # The Counter machine changes state based on the value
4 # of its 'counter' property. Initially the counter
5 # is set to 0 and this causes the start state to
6 # activate. This in turn causes the counter to
7 # increase, activating the next state.
8 
9 Counter MACHINE {
10  start WHEN counter == 0;
11  one WHEN counter == 1;
12  two WHEN SELF.counter == 2;
13  init DEFAULT;
14 
15  ENTER init { LOG "unknown"; counter := 0; }
16 
17  ENTER start { LOG "start"; WAIT 1000; INC counter }
18  ENTER one { LOG "one"; WAIT 1000; INC counter}
19  ENTER two { LOG "two"; WAIT 1000; INC counter}
20  ENTER three { LOG "three"; WAIT 1000; INC counter }
21 }
