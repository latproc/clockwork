1 /* Here is a virtual input to support a piston that moves extends
2  and retracts but has only one sensor,
3  in the middle of the stroke.
4 
5  To use it, try something like:
6 
7  extend_output FLAG (tab:Examples);
8  retract_output FLAG (tab:Examples);
9  i_centre FLAG (tab:Examples);
10 
11  piston_stroke_speed StrokeMovementParameters(speed:1000);
12 
13  extend_monitor StrokeMonitor(tab:Examples) piston_stroke_speed, i_centre, extend_output, retract_output;
14  retract_monitor StrokeMonitor(tab:Examples) piston_stroke_speed, i_centre, retract_output, extend_output;
15 
16  extend_params StrokeParameters(on_distance:4000);
17  retract_params StrokeParameters(on_distance:2000);
18 
19  i_extended MeasuredStroke (tab:Examples) extend_params, extend_monitor;
20  i_retracted MeasuredStroke (tab:Examples) retract_params, i_retracted;
21 
22  extend_monitor StrokeMonitor(tab:Examples)
23 
24 */
25 
26 /* SPEED
27 
28  We want a concervative estimate of when the stroke has
29  completed. To achieve this we assume the piston
30  retracts faster than it extends.
31 
32  The stroke defines micro_position to be the entire range
33  of motion within the proximity range. When the
34  actuator moves away from the reset mark it sets
35  its micro_position to a compensated value to cater for
36  a few milliseconds of latency before it sensed the
37  off transition.
38 */
39 
40 StrokeMovementParameters MACHINE {
41  OPTION speed 1000; # mm per sec to extend
42  OPTION retract_bonus 10; # moves 10mm/sec faster when retracting
43  OPTION start_compensation 2; # expected distance (mm) by the time we detect the input has just turned off\
44  OPTION max_travel 4000;
45  ready INITIAL;
46 }

50 StrokeParameters MACHINE {
51  OPTION on_distance 0;
52 }
53 
54 MeasuredStroke MACHINE settings, monitor {
55  on WHEN monitor.position >= settings.on_distance AND monitor != error;
56  off DEFAULT;
57 }

59 /* MONITOR
60 
61  This stroke monitor that measures the on time of two outputs
62  and determines whether a stroke is complete or not.
63 
64  This virtual input never moves to a negative micro_position
65 
66 */
67 StrokeMonitor MACHINE params, reset_pin, i_extend, i_retract {
68  OPTION position 0;
69  OPTION micro_position 0; # compensate for lack of float
70  OPTION start_time 0;
71  OPTION end_time 0;
72  OPTION speed 0;
73 
74  error WHEN SELF IS error OR i_extend IS on AND i_retract IS on;
75  idle WHEN i_extend IS off AND i_retract IS off;
76  recalc WHEN SELF IS operating AND TIMER > 20;
77  operating WHEN i_extend IS on OR i_retract IS on;
78  unknown DEFAULT; # only possible if the inputs are in unusual states
79 
80  ENTER error { micro_position := 0; position := 0; }
81  RECEIVE i_extend.on_enter {
82  LOG "extend on";
83  start_time := NOW;
84  speed := params.speed;
85  }
86  RECEIVE i_extend.off_enter {
87  CALL recalc_pos ON SELF;
88  speed := 0;
89  }
90 
91  RECEIVE i_retract.on_enter {
92  start_time := NOW;
93  speed := 0 - params.speed - params.retract_bonus;
94  }
95  RECEIVE i_retract.off_enter {
96  CALL recalc_pos ON SELF;
97  speed := 0;
98  }
99 
100  RECEIVE recalc_pos {
101  end_time := NOW;
102  micro_position := micro_position + (end_time - start_time) * speed;
103  IF (reset_pin IS on) { micro_position := 0; }
104  ELSE {
105  IF (micro_position > params.max_travel * 1000) { micro_position := params.max_travel * 1000 }
106  };
107  position := micro_position / 1000;
108  }
109  ENTER recalc {
110  CALL recalc_pos ON SELF;
111  start_time := end_time;
112  }
113 
114  # whenever the reset pin activates we know we are at home micro_position
115  # (ie fully retracted from this input point of view)
116  RECEIVE reset_pin.on_enter {
117  start_time := NOW;
118  micro_position := 0;
119  position := 0;
120  }
121  RECEIVE reset_pin.off_enter {
122  # rather than use an if to allow for direction we use speed / params.speed
123  # as a scalar to the start_componensation.
124  micro_position := speed * params.start_compensation / params.speed;
125  position := micro_position / 1000;
126  }
127 }
